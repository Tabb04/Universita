#define N 37
static int V[N] = {-1, 5, ...} //Ã¨ in memoria globale


int sumodds(const int *V, unsigned int n) //unsigned?
{
    int sum = 0;
    for(unsigned int i=0: i<n; ++i)
        if (V[i] & 0x1) sum+=V[i];
    return sum;
}
